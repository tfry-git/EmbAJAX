606 template<size_t NUM> class EmbAJAXRadioGroup : public EmbAJAXContainer<NUM>, public EmbAJAXRadioGroupBase {
607 public:
613  EmbAJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : EmbAJAXContainer<NUM>(), EmbAJAXRadioGroupBase() {
614  for (uint8_t i = 0; i < NUM; ++i) {
615  char* childid = childids[i];
616  strncpy(childid, id_base, EMBAJAX_MAX_ID_LEN-4);
617  itoa(i, &(childid[strlen(childid)]), 10);
618  buttons[i] = EmbAJAXCheckButton(childid, options[i], i == selected_option);
619  buttons[i].radiogroup = this;
620  buttonpointers[i] = &buttons[i];
621  }
622  _current_option = selected_option;
623  EmbAJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify EmbAJAXContainer<NUM>::, explicitly?
624  _name = id_base;
625  }
627  void selectOption(uint8_t num) {
628  for (uint8_t i = 0; i < NUM; ++i) {
629  buttons[i].setChecked(i == num);
630  }
631  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
632  }
634  uint8_t selectedOption() const {
635  return _current_option;
636  }
639  EmbAJAXBase* button(uint8_t num) {
640  if (num < NUM) return (&buttons[num]);
641  return 0;
642  }
643 private:
644  EmbAJAXCheckButton buttons[NUM];
645  EmbAJAXBase* buttonpointers[NUM];
646  char childids[NUM][EMBAJAX_MAX_ID_LEN];
647  int8_t _current_option;
648  void selectButton(EmbAJAXCheckButton* which) override {
649  _current_option = -1;
650  for (uint8_t i = 0; i < NUM; ++i) {
651  if (which == buttonpointers[i]) {
652  _current_option = i;

654  buttons[i].setChecked(false);

656  }
